### 💻 Appendix [L260504a]: Teleportation Logic
*(ข้อมูลอ้างอิงของแล็บ: [Level 3: Event-Driven Teleportation System](#-l260504a-level 3-event-driven-teleportation-system))*

```lua
-- [[Lab-ID: L260504a]]
local teleportPad = script.Parent
local destination = game.Workspace.DestinationPad -- พิกัดระบุตำแหน่งเป้าหมายปลายทาง

local debounce = false -- การจัดการสถานะตัวแปร (State Management) เพื่อป้องกันการประมวลผลซ้อนกัน

teleportPad.Touched:Connect(function(hit)
    if not debounce then
        local character = hit.Parent
        local humanoid = character:FindFirstChild("Humanoid")
        
        if humanoid then
            debounce = true
            local rootPart = character:FindFirstChild("HumanoidRootPart")
            
            if rootPart then
                -- แก้ไขบั๊กตัวละครจมตกแมพ (Floor Clipping Error) ด้วยการบวก Vector3 Offset เข้ากับแกน Y
                rootPart.CFrame = destination.CFrame + Vector3.new(0, 5, 0)
            end
            
            task.wait(2) -- คำสั่งหน่วงเวลาคูลดาวน์ระบบ 2 วินาที (Cooldown Protocol)
            debounce = false
        end
    end
end)
^L260504a
```

### 💻 Appendix [L260511a]: Access Control Gate Logic
*(ข้อมูลอ้างอิงของแล็บ: [Level 2: Conditional Access Control Gate](#-l260511a-level 2-conditional-access-control-gate))*

```lua
-- [[Lab-ID: L260511a]]
local gatePart = script.Parent
local triggerPad = game.Workspace.SecurityTrigger

local isOpen = false -- ประกาศตัวแปรเริ่มต้นสลับสถานะ (State Variable)

triggerPad.Touched:Connect(function(hit)
    local character = hit.Parent
    local humanoid = character:FindFirstChild("Humanoid")
    
    -- การประเมินผลเงื่อนไขคู่: ตรวจเจอโมเดลผู้เล่นจริง AND ประตูบานนี้ต้องมีสถานะเป็นปิดอยู่เท่านั้น
    if humanoid and not isOpen then
        isOpen = true
        gatePart.Transparency = 0.8 -- ปรับความโปร่งแสงจำลองว่าประตูกำลังเปิด
        gatePart.CanCollide = false  -- ปิดกล่องชนเพื่อให้ผู้เล่นเดินทะลุผ่านไปได้
        
        task.wait(3) -- คำสั่งเปิดประตูทิ้งไว้ให้เดิน 3 วินาที
        
        gatePart.Transparency = 0 -- สั่งปิดประตู ปรับกลับมาทึบแสงเหมือนเดิม
        gatePart.CanCollide = true  -- เปิดกล่องชนเพื่อสร้างกำแพงกักกันความปลอดภัยอีกครั้ง
        isOpen = false
    end
end)
^L260511a
```
